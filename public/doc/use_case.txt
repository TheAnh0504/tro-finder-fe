@startuml
skinparam conditionStyle Inside
title Quy trình nghiệp vụ tìm phòng và ký kết hợp đồng khép kín

|Bên khởi tạo (Chủ trọ / Người thuê)|
start
:Tìm kiếm và thảo luận điều khoản qua Khung Chat;
:Chọn chức năng Tạo hợp đồng điện tử;

|Hệ thống|
:Truy xuất dữ liệu định danh OCR sẵn có;
:Tự động điền thông tin cá nhân của hai bên;

|Bên khởi tạo (Chủ trọ / Người thuê)|
:Nhập giá thuê, tiền cọc và thời hạn;
if (Lựa chọn thời điểm ký hợp đồng?) then (Ký trước khi gửi)
    :Thực hiện Ký xác nhận hợp đồng;
    |Hệ thống|
    :Lưu hợp đồng (Trạng thái: Khởi tạo đã ký);
else (Gửi bản nháp rà soát sau)
    |Bên khởi tạo (Chủ trọ / Người thuê)|
    :Nhấn nút Gửi bản nháp hợp đồng;
    |Hệ thống|
    :Lưu hợp đồng (Trạng thái: Chờ rà soát);
endif

:Gửi thông báo đến tài khoản Bên tiếp nhận;

|Bên tiếp nhận (Người thuê / Chủ trọ)|
:Truy cập không gian làm việc số để rà soát hợp đồng;
if (Đồng ý với các điều khoản?) then (Có)
    :Thực hiện Ký xác nhận hợp đồng;
    |Hệ thống|
    if (Hợp đồng đã thu thập đủ 2 chữ ký?) then (Chưa đủ (Bên khởi tạo chưa ký))
        :Cập nhật trạng thái (Chờ bên khởi tạo ký);
        |Bên khởi tạo (Chủ trọ / Người thuê)|
        :Nhận thông báo phản hồi và thực hiện Ký xác nhận;
        |Hệ thống|
    else (Đã đủ 2 chữ ký)
    endif
    :Hợp đồng chính thức chuyển sang trạng thái "Hoàn thành";
    :Lưu trữ tài liệu hợp đồng điện tử an toàn;
    :Cập nhật trạng thái phòng sang "Đã cho thuê";
    stop
else (Không đồng ý)
    |Bên tiếp nhận (Người thuê / Chủ trọ)|
    :Nhấn Từ chối ký và yêu cầu hủy bỏ/sửa đổi;
    |Hệ thống|
    :Cập nhật trạng thái hợp đồng thành "Đã hủy";
    stop
endif
@enduml

service ..> utils : helper

repository ..> entity : manages
service ..> entity : processes
